◈ REPLAY DATA FORMAT

Stored as a simple JSON array — 20 bytes per turn. Far smaller than frame-by-frame recording.
interface TurnRecord {
  player: 'hider' | 'seeker';
  aimAngle: number; // radians
  aimPower: number; // 0-1
  startX: number;
  startY: number;
  timestamp: number; // ms
  tagged?: boolean;
}
